\block{\textcolor{white}{\large \ding{202}\ \ Introduction}}{
    \small
    \textbf{Challenge:} Radar-based person identification faces two critical hurdles:
    \begin{itemize}
        \setlength\itemsep{2pt}
        \item \textbf{Sparse/noisy point clouds} – motion defocusing and multipath ghosts
        \item \textbf{Data scarcity} – requires $>$100 sequences per person
    \end{itemize}
    \vspace{3mm}
    \textbf{Our Solution:} Unified framework combining:
    \begin{itemize}
        \setlength\itemsep{2pt}
        \item \textbf{InISAR:} High-quality 3D reconstruction with multipath suppression
        \item \textbf{PCTtrans + Pre-trained MAML:} Few-shot learning from minimal data
    \end{itemize}
    \vspace{3mm}
    \textbf{Impact:} 70.0\% cross-session accuracy with just 10 examples per person
}

\block{\textcolor{white}{\large \ding{209}\ \ Pre-trained MAML}}{
    \small
    \textbf{Two-Phase Algorithm:}
    \begin{enumerate}
        \setlength\itemsep{2pt}
        \item \textbf{Phase 1:} Supervised training on representative task (3 subjects) $\to$ $\theta_0^{PT}$
        \item \textbf{Phase 2:} Meta-train MAML with $\theta_0^{PT}$ initialization
    \end{enumerate}
    \vspace{3mm}
    \begin{align*}
        \text{Inner Loop: } & \theta_i' = \theta - \alpha \nabla \mathcal{L}_{\mathcal{T}_i} \\
        \text{Outer Loop: } & \theta \leftarrow \theta - \beta \nabla \sum_i \mathcal{L}_{\mathcal{T}_i}
    \end{align*}
    \small
    \textbf{Key Advantage:} $\theta_0^{PT}$ is closer to optimal $\theta^*$ than $\theta_0^{rand}$,
    reducing meta-training iterations.
}
