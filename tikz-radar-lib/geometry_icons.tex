% ---------------------------------------------------------------
% \CoordinateFrame -- xyz axes triad
% ---------------------------------------------------------------
\newcommand{\CoordinateFrame}{%
  \begin{scope}
    \draw[geometryColor, -{Latex[length=2mm]}, line width=0.9pt] (-0.3,0) -- (0.3,0) node[right, font=\tiny] {x};
    \draw[geometryColor, -{Latex[length=2mm]}, line width=0.9pt] (0,-0.3) -- (0,0.3) node[above, font=\tiny] {y};
    \draw[geometryColor, -{Latex[length=2mm]}, line width=0.9pt] (0,0) -- (-0.25,-0.25) node[below left, font=\tiny] {z};
    \fill[iconStroke] (0,0) circle (0.02);
  \end{scope}
}

% ---------------------------------------------------------------
% \VoxelCube -- 3D occupancy grid
% ---------------------------------------------------------------
\newcommand{\VoxelCube}{%
  \begin{scope}[scale=0.7]
    \draw[iconStroke, fill=white, rounded corners=2pt] (-0.55,-0.55) rectangle (0.55,0.55);
    \coordinate (dx) at (0.2,0);
    \coordinate (dy) at (0,0.22);
    \coordinate (dz) at (0.12,0.08);
    \foreach \i in {0,1,2}{
      \foreach \j in {0,1}{
        \pgfmathsetmacro{\shade}{30+20*\i}
        \draw[iconStroke, line width=0.4pt, fill=geometryColor!\shade!white, fill opacity=0.6]
          ($(-0.4,-0.35)+\i*(dx)+\j*(dy)$) --
          ($(-0.4,-0.35)+(\i+1)*(dx)+\j*(dy)$) --
          ($(-0.4,-0.35)+(\i+1)*(dx)+(\j+1)*(dy)$) --
          ($(-0.4,-0.35)+\i*(dx)+(\j+1)*(dy)$) -- cycle;
      }
    }
    \draw[iconStroke, line width=0.6pt] (-0.4,-0.35) -- ($(-0.4,-0.35)+(dz)$);
    \draw[iconStroke, line width=0.6pt] (0.2,-0.35) -- ($(0.2,-0.35)+(dz)$);
    \draw[iconStroke, line width=0.6pt] (0.2,0.09) -- ($(0.2,0.09)+(dz)$);
    \draw[iconStroke, line width=0.6pt] (-0.4,0.09) -- ($(-0.4,0.09)+(dz)$);
  \end{scope}
}

% ---------------------------------------------------------------
% \ICPRegistration -- aligning two point clouds with correspondences
% ---------------------------------------------------------------
\newcommand{\ICPRegistration}{%
  \begin{scope}
    \draw[iconStroke, fill=white, rounded corners=2pt] (-0.55,-0.55) rectangle (0.55,0.55);
    % Source (faint)
    \foreach \p in {(-0.4,0.25),(-0.35,-0.05),(-0.4,-0.25),(-0.15,0.1)}{
      \fill[iconFaint] \p circle (0.035);
    }
    % Target (colored)
    \foreach \p in {(-0.25,0.3),(-0.2,0.0),(-0.3,-0.2),(-0.05,0.15)}{
      \fill[geometryColor] \p circle (0.04);
    }
    % Correspondences (dashed)
    \draw[iconFaint, line width=0.4pt, dashed] (-0.4,0.25) -- (-0.25,0.3);
    \draw[iconFaint, line width=0.4pt, dashed] (-0.35,-0.05) -- (-0.2,0.0);
    \draw[iconFaint, line width=0.4pt, dashed] (-0.4,-0.25) -- (-0.3,-0.2);
    % Arrow and aligned set
    \draw[geometryColor, -{Latex[length=2mm]}, line width=0.8pt] (0.0,0.0) -- (0.15,0.0);
    \foreach \p in {(0.32,0.2),(0.4,-0.05),(0.35,-0.25),(0.2,0.1)}{
      \fill[geometryColor] \p circle (0.04);
    }
  \end{scope}
}

% ---------------------------------------------------------------
% \RigidTransform -- rotation/translation arrow between point sets
% ---------------------------------------------------------------
\newcommand{\RigidTransform}{%

    \draw[iconStroke, fill=white, rounded corners=2pt] (-0.55,-0.55) rectangle (0.55,0.55);
    % Source points (faint)

      \fill[iconFaint] \p circle (0.03);
    }
    % Transform matrix symbol
    \node[draw=geometryColor, fill=geometryColor!15, rounded corners=1pt,
          minimum width=0.6cm, minimum height=0.4cm, font=\tiny\bfseries] (T) at (0,0) {R,T};
    % Target points (colored)
    \foreach \p in {(0.25,0.25),(0.35,0.05),(0.25,-0.15),(0.4,-0.2),(0.35,0.15)}{
      \fill[geometryColor] \p circle (0.035);

    \draw[geometryColor, -{Latex[length=1.5mm]}, line width=0.5pt] (-0.28,0.1) -- (T.west);
    \draw[geometryColor, -{Latex[length=1.5mm]}, line width=0.5pt] (T.east) -- (0.28,0.1);
